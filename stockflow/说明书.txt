# 1) 初始化数据库（首次运行任何命令都会自动建库+迁移）
python -m stockflow.ui.cli product-list

# 2) 创建一个仓库
python -m stockflow.ui.cli wh-add --code=MAIN --name=主仓

# 3) 新增商品 + 入库 + 查询库存
python -m stockflow.ui.cli product-add --sku=J001 --name=18K戒指 --spec=Au750 --unit=pcs --cost=1200 --price=1680
# 假设商品ID为 1、仓库ID为 1（刚生成的）
python -m stockflow.ui.cli inbound --product-id=1 --wh-id=1 --qty=5
python -m stockflow.ui.cli stock --product-id=1 --wh-id=1
